 فایل‌های پیکربندی
/etc/systemd/system/startupdesign.service


[Unit]
Description=StartupDesign Django App (Gunicorn)
After=network.target

[Service]
User=root
Group=root
WorkingDirectory=/srv/startup_design_django
ExecStart=/srv/startup_design_django/venv/bin/gunicorn \
    --workers 3 \
    --bind unix:/run/startupdesign.sock \
    --access-logfile /var/log/startupdesign/access.log \
    --error-logfile /var/log/startupdesign/error.log \
    StartupDesign.wsgi:application
Restart=always
RestartSec=5

[Install]
WantedBy=multi-user.target






/etc/nginx/sites-available/startupdesign

server {
    listen 80;
    server_name your-domain.com;  # یا IP سرور

    access_log /var/log/nginx/startupdesign_access.log;
    error_log  /var/log/nginx/startupdesign_error.log;

    location /static/ {
        alias /srv/startup_design_django/staticfiles/;
    }

    location /media/ {
        alias /srv/startup_design_django/media/;
    }

    location / {
        include proxy_params;
        proxy_pass http://unix:/run/startupdesign.sock;
    }
}
